### Code Block

```typescript
interface User {
  id: string
  name: string
  email: string
}

function greetUser(user: User): string {
  return `Hello, ${user.name}!`
}

const user: User = {
  id: "1",
  name: "Kevin",
  email: "kevin@example.com",
}

console.log(greetUser(user))
```

### Code Block with Different Language

```python
def fibonacci(n: int) -> int:
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

for i in range(10):
    print(fibonacci(i))
```
